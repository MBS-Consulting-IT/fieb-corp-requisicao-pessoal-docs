  monthlyAlert,
  hourlyAlert,
  internAlert

const WorkloadType = function () {

    position: document.querySelector('[xname=inpcargoRP]'),
    positionId: document.querySelector('[xname=inpcargoRPCodigo]'),
    workloadBase: document.querySelector('[xname=inpcargaHorariaBase]')
  }

  const state = {
    emitter: null,
    isHourly: null,
    isWeek: null,
    isProportional: null,
    hourlyTypes: [
      'Horista CTD',
      'Horista CTI'
    ],
    internTypes: [
      'Est√°gio'
    ],
    proportionalTypes: [
      'Horista Intermitente'
    ],
    hourlyWorkload: {
      workload: [{ min: 0, max: 20 }],
      alert: hourlyAlert,
      message: 'at√© 20h'
    },
    internWorkload: {
      workload: [{
        min: 20, max: 20
      }, {
        min: 30, max: 30
      }],
      alert: internAlert,
      message: '20h ou 30h'
    },
    monthlyWorkload: {
      workload: [{
        min: 20, max: 20
      }, {
        min: 40, max: 40
      }],
      alert: monthlyAlert,
      message: '20h ou 40h'
    },
    positionsWorload: WorkloadPositions
  }

  return {
    mount
  }

  /**
   * üîë M√©todos P√∫blicos

  /**

   * @public
   * @memberof module:WorkloadType
   * @param {object} config.emitter - inst√¢ncia global do event emitter

  function mount ({ emitter }) {
    state.emitter = emitter

    addTriggers()
    handleWorkload()
  }

  /**
   * üîí M√©todos Privados
   */

  function addTriggers () {
    /**
     * Utilizando jQuery nos triggers para reagir a mudan√ßas em campos do tipo `sugest√£o`
     */
    $($refs.positionId)
      .on('change', handleWorkload)

    $refs.contractType
      .forEach(field =>
        field.addEventListener('change', handleWorkload)
      )
  }

  function getType () {
    const checked = [...$refs.contractType]
      .find(inp => inp.checked)

    return checked
      ? checked.value
      : null
  }

  function getWorkload (positionId) {
    let workload = {}

    const type = getType()

    const isProportional = state.proportionalTypes
      .includes(type)

    const isHourly = state.hourlyTypes
      .includes(type)

    const isIntern = state.internTypes
      .includes(type)

    if (isIntern) {
      workload = state.internWorkload
    } else if (!isHourly) {
      workload = getPositionWorkload(positionId)
    }

    return {
      type,
      isProportional,
      isHourly,
      isIntern,
      isTeacher: workload?.type === 'teacher',
      ...workload
    }
  }

  function getPositionWorkload (positionId) {
    const workload = state.positionsWorload
      .find(({ positions }) =>
        positions.includes(positionId)
      ) || state.monthlyWorkload

    return workload
  }

  function handleWorkload () {
    const positionId = $refs.positionId.value
    const currentWorkload = getWorkload(positionId)

    state.isProportional = currentWorkload.proportionalTypes
    state.isHourly = currentWorkload.isHourly

    setWeekMode(currentWorkload.isTeacher)

    state.isHourly
      ? setHourlyWorkload(currentWorkload)
      : setMonthlyWorkload(currentWorkload)
  }

  function setWeekMode (isTeacher) {
    const isWeek = !isTeacher

    if (state.isHourly) {
      state.isWeek = true
    } else if (state.isWeek !== isWeek) {
      state.isWeek = isWeek
    }

    return emitWorkloadWeek({ isWeek: state.isWeek })
  }

  function setMonthlyWorkload (currentWorkload) {
    const {
      alert,
      workload,
      message
    } = currentWorkload

    setBaseWorkload(message)

    /**
     * Adicionado delay para previnir capturar o valor do campo `sugest√£o`
     * sem o valor definido. Sem workaround para capturar diretamente.
     */
    setTimeout(() => {
      emitWorkloadChange({
        ...currentWorkload,
        alert: alert(workload, $refs.position.value)
      })
    }, 100)
  }

  function setHourlyWorkload (currentWorkload) {
    const { hourlyWorkload } = state

    setBaseWorkload(hourlyWorkload.message)

    emitWorkloadChange({
      ...currentWorkload,
      ...hourlyWorkload,
      alert: hourlyWorkload.alert()
    })
  }

  function emitWorkloadChange (currentWorkload) {
    const { isHourly, isProportional, isIntern, alert, workload } = currentWorkload

    state.emitter.emit('workload-limit', { workload, alert })
    state.emitter.emit('workload-type', { isHourly, isProportional, isIntern })
    state.emitter.emit('workload-update')
  }

  function emitWorkloadWeek ({ isWeek }) {
    state.emitter.emit('workload-week', { isWeek })
    state.emitter.emit('workload-update')
  }

  function setBaseWorkload (value) {
    $refs.workloadBase.value = value
    $refs.workloadBase.dispatchEvent(new Event('change'))
  }
}

export default WorkloadType()
