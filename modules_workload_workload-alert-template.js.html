    /**
 * Define os templates do conteúdo da mensagem de alerta localizada logo abaixo
 * da tabela de Carga Horária Semanal.
 *
 * @module WorkloadAlertTemplate
 */

/**
 * Alerta destinado aos cargos mensalistas de carga horária de 20h ou 40h semanais
 * @method
 * @returns {string} conteúdo HTML da mensagem
 */
export const monthlyAlert = () => `
<p><strong>Atenção!</strong> Cargos mensalistas devem ter carga horária de <strong>20h</strong> ou <strong>40h</strong>.</p>
`

/**
 * Alerta destinado aos cargos horistas de carga horária até 20h semanais
 * @method
 * @returns {string} conteúdo HTML da mensagem
 */
export const hourlyAlert = () => `
<p><strong>Atenção!</strong> Cargos horistas devem ter carga horária de até <strong>20h</strong>.</p>
`

/**
 * Alerta destinado aos cargos de estágio
 * @method
 * @returns {string} conteúdo HTML da mensagem
 */
export const internAlert = (workload) => {
  const [start, end] = workload

  return `
    <p><strong>Atenção!</strong> Estagiários podem ter carga horária de <strong>${start.max}h</strong> ou <strong>${end.max}h</strong>.</p>
  `
}

/**
 * Alerta destinado aos cargos de professor
 * @method
 * @returns {string} conteúdo HTML da mensagem
 */
export const teacherAlert = (workload) => `
<p><strong>Atenção!</strong> Professores podem ter carga horária de até <strong>${workload[0].max}h</strong>.</p>
`

/**
 * Alerta destinado aos cargos de carga horária específica com **múltiplas** faixas de carga horária
 * @method
 * @returns {string} conteúdo HTML da mensagem
 */
export const multipleJourneyAlert = (workload, position) => {
  const journeysMarkup = workload.map(({ min, max }) => {
    if (min === max) {
      return `<strong>${min}h</strong>`
    }

    return `<strong>${min}h a ${max}</strong>`
  })

  return `
  <p><strong>Atenção!</strong> ${position} deve ter carga horária de ${journeysMarkup.join(' ou ')}.</p>
  `
}

/**
 * Alerta destinado aos cargos de carga horária específica com **única** faixa de carga horária
 * @method
 * @returns {string} conteúdo HTML da mensagem
 */
export const positionAlert = (workload, position) => {
  const min = workload[0].min
  const max = workload[0].max

  if (min === max) {
    return `
      <p><strong>Atenção!</strong> ${position} deve ter carga horária de <strong>${max}h</strong>.</p>
    `
  }

  return `
    <p><strong>Atenção!</strong> ${position} deve ter carga horária de <strong>${min}h</strong> a <strong>${max}h</strong>.</p>
  `
}
