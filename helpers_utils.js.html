    /**
 * @module utils
 * @description Funções utilitárias para uso geral
 */

export const currencyFormatter = new Intl.NumberFormat('pt-BR', {
  minimumFractionDigits: 2,
  maximumFractionDigits: 2
})

/**
 * Verifica se uma string respeita o formato de horário HH:MM
 * @param {string} time horário no formato HH:MM
 * @returns {boolean}
 */
export function isValidTime (time) {
  const [HH, MM] = time.split(':')

  return !isNaN(
    Date.parse(`2000-01-01T${HH}:${MM}:00`)
  )
}

/**
 * Verifica se uma string respeita o formato de duração em HH:MM,
 * limitando a duração em horas até 99h
 * @param {string} duration duração no formato [HH]:MM
 * @returns {boolean}
 */
export function isValidDuration (duration) {
  const MM = duration.split(':')[1]

  return !(!MM || MM.length !== 2 || MM > parseInt(59))
}

/**
 * Calcula a diferença entre datas em milissegundos
 * @param {Date} start data inicial
 * @param {Date} end data final
 * @returns {number} diferença em milissegundos
 */
export function getDifferenceInMs (start, end) {
  return end.getTime() - start.getTime()
}

/**
 * Obtém um objeto data de uma string horário no formato HH:MM
 * @param {string} time horário no formato HH:MM
 * @returns {?Date}
 */
export function getDateFromTime (time) {
  const isValid = isValidTime(time)

  if (!isValid) {
    return null
  }

  const [HH, MM] = time.split(':')

  return new Date(`2000-01-01T${HH}:${MM}:00`)
}

/**
 * @typedef {Object} Duration
 * @property {string} mask máscara em [HH]:MM
 * @property {number} hour quantidade de horas
 * @property {number} minutes quantidade de minutos
 * @property {number} seconds quantidade de segundos
 * @property {number} milliseconds quantidade de milissegundos
 */

/**
 * Calcula determinada duração em horas, minutos, segundos
 * e minutos separadamente. Além de informar a duração com a
 * máscara [HH]:MM
 *
 * @param {string} duration duração em milissegundos
 * @returns {Duration}
 */
export function msToTime (duration) {
  const hours = Math.floor((duration / (1000 * 60 * 60)))
  const minutes = Math.floor((duration / (1000 * 60)) % 60)
  const seconds = Math.floor((duration / 1000) % 60)
  const milliseconds = parseInt((duration % 1000) / 100)

  return {
    mask: `${timePadZero(hours)}:${timePadZero(minutes)}`,
    hours,
    minutes,
    seconds,
    milliseconds
  }
}

/**
 * Converte uma string no formato [HH]:MM em decimal
 * @param {string} duration duração no formato [HH]:MM
 * @returns {?number}
 */
export function durationToDecimal (duration) {
  const isValid = isValidDuration(duration)

  if (!isValid) {
    return null
  }

  const [HH, MM] = duration.split(':')

  return parseInt(HH) + parseInt(MM) / 60
}

/**
 * Adiciona `0` em horários menores que 10
 * @example 5 => 05
 * @param {string} time tempo
 * @returns {string}
 */
export function timePadZero (time) {
  return (time).toString().padStart(2, '0')
}

/**
 * Transforma uma string monetária em número.
 * Quando informado strings vazias o valor 0 é retornado
 * @example '1.000,50' => 1000.5
 * @param {string} currencyValue valor monetário pt-BR
 * @returns {number}
 */
export function currencyToNumber (currencyValue) {
  return parseFloat(
    currencyValue
      .replace(/\./g, '')
      .replace(',', '.') || 0
  )
}

/**
 * Verifica se o valor está no formato monetário
 * @param {string|number} value - valor
 * @returns {boolean}
 */
export function isCurrency (value) {
  if (typeof value !== 'string') {
    return false
  }

  const [int, frac] = (value).split(',')
  const isValidFormat = (frac || '').length === 2 && !isNaN(parseInt(int))

  return isValidFormat
}

/**
 * Transforma um número em valor monetário pt-BR
 * @param {number} number valor
 * @returns {string} valor monetário
 */
export function numberToCurrency (number) {
  return currencyFormatter.format(number)
}

/**
 * Altera o valor de um input disparando o evento de `change`
 * @param {HTMLInputElement} input
 * @param {String|Number} value valor a ser modificado
 */
export function setValue (input, value) {
  input.value = value
  input.dispatchEvent(new Event('change'))
}

/**
 * Altera a formatação de um texto, deixando somente as letras
 * iniciais maiúsculas.
 *
 * @example
 * - SOU UM ITEM
 * - SOU OUTRO ITEM
 *
 * - Sou um item
 * - Sou outro item
 *
 * @param {string} text lista para formatação
 * @returns {string} lista formatada
 */
export function formatListCase (text = '') {
  const formatted = text
    .split('- ')
    .map(item =>
      item.charAt(0).toUpperCase() +
      item.slice(1).toLowerCase()
    )
    .join('- ')

  return formatTextBreaks(formatted)
}

/**
 * Formata quebra de linhas para o formato `\n` exclusivamente
 * @param {string} text texto para formatação
 * @returns {string} texto formatado
 */
export function formatTextBreaks (text = '') {
  return text
    .replace(/\\r\\n$/, '')
    .replaceAll('\\r\\n', '\n')
    .trim()
}
