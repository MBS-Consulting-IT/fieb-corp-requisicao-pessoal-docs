    import Progress from '../../components/progress/index'
import { durationToDecimal, msToTime } from '../../helpers/utils'
import { getCurrentLimit, isValidRange } from './workload-utils'

/**
 * Controla as validações e o progresso da tabela de carga horária, intervindo com o
 * bloqueio/desbloqueio dos botões de início conforme o intervalos permitidos.
 *
 * @module WorkloadLimiter
 */
const WorkloadLimiter = function () {
  const $refs = {
    progress: document.querySelector('#workload-progress'),
    alert: document.querySelector('#workload-alert'),
    alertContent: document.querySelector('#workload-alert .o-alert-content'),
    limitLabel: document.querySelector('#workload-limit'),
    workloadBase: document.querySelector('[xname=inpcargaHorariaBase]')
  }

  /**
   * @memberof module:WorkloadLimiter
   * @property {object} validator - instância do componente `TaskValidator`
   * @property {number} total - carga horária total em formato decimal
   * @property {number|object} currentLimit - intervalo ativo de carga horária
   * @property {object[]} limits - intervalos de carga horária
   */
  const state = {
    validator: null,
    total: 0,
    currentLimit: 0,
    limits: [{ min: null, max: null }]
  }

  /**
   * Representação de 1h em milissegundos
   * @memberof module:WorkloadLimiter
   * @type {number}
   */
  const HOUR_IN_MS = 3600000

  /**
   * Tipos de erros utilizados no componente `TaskValidator`
   * @memberof module:WorkloadLimiter
   * @type {object}
   */
  const VALIDATOR_ERRORS = {
    filling: 'workload_filling',
    limit: 'workload_limit_exceeded'
  }

  /**
   * Instância do componente `Progress`
   * @memberof module:WorkloadLimiter
   * @type {object}
   */
  const workloadProgress = Progress($refs.progress, { max: 0 })

  return {
    mount,
    limit,
    update
  }

  /**
   * 🔑 Métodos Públicos
   */

  /**
   * Realiza a montagem do módulo
   * @public
   * @memberof module:WorkloadLimiter
   * @param {object} config.emitter - instância global do event emitter
   * @param {object} config.validator - instância global do validator
   */
  function mount ({ emitter, validator }) {
    state.emitter = emitter
    state.validator = validator
  }

  /**
   * Define novas faixas de carga horária
   * @public
   * @memberof module:WorkloadLimiter
   * @param {object[]} config.workload - faixas de carga horária
   * @param {string} config.alert - html da mensagem de alerta de carga horária extrapolada
   */
  function limit ({ workload, alert }) {
    state.limits = workload

    handleLimits()
    updateAlert(alert)
  }

  /**
   * Atualiza o total da carga horária no estado do componente,
   * validando o intervalo permitido e atualizando a barra de progresso.
   * @public
   * @memberof module:WorkloadLimiter
   * @param {number} value - carga horária total
   */
  function update (value) {
    state.total = durationToDecimal(value)

    handleLimits()
  }

  /**
   * 🔒 Private Methods
   */

  function handleLimits () {
    const previousLimit = state.currentLimit
    const currentLimit = getCurrentLimit(state.total, state.limits)
    const isAboveLimit = state.limits
      .every(({ max }) => state.total > max)

    state.currentLimit = currentLimit || state.currentLimit

    if (
      state.currentLimit.max !== previousLimit.max ||
      state.currentLimit.min !== previousLimit.min
    ) {
      workloadProgress.limits(state.currentLimit)
      updateLimitLabel(state.currentLimit.max)
    }

    if (isValidRange(state.total, state.currentLimit)) {
      removeWarnings()
    } else {
      isAboveLimit
        ? addWarningAboveLimit()
        : addWarningFilling()
    }

    updateProgress()
  }

  function updateProgress () {
    workloadProgress.update(state.total)
  }

  function updateAlert (content) {
    $refs.alertContent.innerHTML = content
  }

  function updateLimitLabel (limit) {
    const limitMask = msToTime(limit * HOUR_IN_MS).mask
    $refs.limitLabel.textContent = limitMask
  }

  function addWarningFilling () {
    const workload = $refs.workloadBase.value
    const type = VALIDATOR_ERRORS.filling
    const unitPosfix = isNaN(workload) ? '' : 'h'
    const message = $refs.workloadBase.value !== ''
      ? `A carga horária semanal deve ser preenchida até o valor de ${workload}${unitPosfix}.`
      : 'A carga horária semanal deve ser preenchida até atingir a carga horária base.'

    state.validator
      .removeErrors([VALIDATOR_ERRORS.limit])

    state.validator
      .addErrors([{ type, message }])
  }

  function addWarningAboveLimit () {
    const limit = state.currentLimit.max
    const current = msToTime(state.total * HOUR_IN_MS).mask
    const type = VALIDATOR_ERRORS.limit
    const message = `A carga horária semanal não pode exceder o total de ${limit}h. O valor total atual é ${current}`

    state.validator
      .removeErrors([VALIDATOR_ERRORS.filling])

    state.validator
      .addErrors([{ type, message }])
  }

  function removeWarnings () {
    state.validator
      .removeErrors(Object.values(VALIDATOR_ERRORS))
  }
}

export default WorkloadLimiter()
