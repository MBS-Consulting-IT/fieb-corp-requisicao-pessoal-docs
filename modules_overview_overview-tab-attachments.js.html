    import { createTableRowsByFields } from './overview-common'

/**
 * Realiza a montagem da aba `Anexos`
 *
 * @module OverviewTabAttachments
 */
const OverviewTabAttachments = function () {
  const $refs = {
    requestAttachmentsTable: document.querySelector('#tbl-attachments-tab'),
    requestAttachments: document.querySelectorAll('[xname=inpanexoArquivo]'),
    requestAttachmentsText: document.querySelectorAll('[xname=inpanexoDescricao]'),
    mailAttachmentsTable: document.querySelector('#tbl-attachments-mail-tab'),
    mailAttachments: document.querySelectorAll('[xname=inpemailAnexo]'),
    mailAttachmentsText: document.querySelectorAll('[xname=inpemailAnexoDescricao]')
  }

  return {
    mount
  }

  /**
   * üîë M√©todos P√∫blicos
   */

  /**
   * Realiza a montagem do m√≥dulo
   * @public
   * @memberof module:OverviewTabAttachments
   */
  function mount () {
    mountAttachmentsTable({
      table: $refs.requestAttachmentsTable,
      fields: [
        $refs.requestAttachments,
        $refs.requestAttachmentsText
      ]
    })

    mountAttachmentsTable({
      table: $refs.mailAttachmentsTable,
      fields: [
        $refs.mailAttachments,
        $refs.mailAttachmentsText
      ]
    })
  }

  /**
   * üîí M√©todos Privados
   */

  function isTableEmpty (fields) {
    return [...fields].every(field => !field.value)
  }

  function mountAttachmentsTable ({ table, fields }) {
    const [
      attachments,
      descriptions
    ] = fields

    const isEmpty = isTableEmpty(attachments)

    if (isEmpty) {
      return
    }

    const columns = [
      { fields: attachments, cellTemplate: attachmentAnchorTemplate },
      { fields: descriptions }
    ]
    const tbody = table.querySelector('tbody')
    const rows = createTableRowsByFields(columns)

    table.classList.remove('u-hidden')
    tbody.insertAdjacentHTML('afterbegin', rows)
  }
}

export default OverviewTabAttachments()

const attachmentAnchorTemplate = value => {
  if (value === null || value === '') {
    return '<td><em>Nenum anexo</em></td>'
  }

  const filename = value
    .split('//')
    .pop()

  const [name, extension] = filename.split('.')
  const displayName = name.length > 18
    ? `${name.substring(0, 18)}...`
    : name

  return `<td><a href="${value}">${displayName}${extension}</a></td>`
}
