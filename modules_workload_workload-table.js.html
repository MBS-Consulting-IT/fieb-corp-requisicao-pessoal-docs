  function validateFormat (inputs) {

      inputs.map(input => inputError.add(input, error))
      throw new Error(error)
    }

    return false
  }

  function handleUpdate (skipUpdate) {
    if (skipUpdate) return
    updateTotal()
  }

  function getTotal () {
    const weekModifier = state.isWeek ? 5 : 1

    return state.workloadTable
      .getRows()
      .map(row => {
        const [start, end] = getRowInputs(row)
          .map(input => getDateFromTime(input.value))

        return start && end
          ? getDifferenceInMs(start, end)
          : 0
      })
      .reduce((total, rowTotal) => rowTotal + total, 0) * weekModifier
  }

  function updateTotal () {
    const total = getTotal()
