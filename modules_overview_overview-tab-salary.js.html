    import { currencyToNumber } from '../../helpers/utils'
import {
  isSalaryAboveMidPoint,
  createTableRowsByFields,
  getPercentDistance,
  assesmentLabelTemplate
} from './overview-common'

/**
 * Realiza a montagem da aba `Remuneração`
 *
 * @module OverviewTabSalary
 */
const OverviewTabSalary = function () {
  let $refs = {}

  /**
   * Busca os campos no momento de execução para evitar
   * carregar de forma incompleta na página de relatório,
   * onde o carregamento acontece de forma assíncrona
   */
  const getRefs = () => ({
    salaryAlertMessage: document.querySelector('#salary-alert-message'),
    assesmentBarChart: document.querySelector('#assessment-bar-chart'),
    workloadTable: document.querySelector('#tbl-workload-tab'),
    assessmentTable: document.querySelector('#tbl-assessment-tab'),
    salaryStepInner: document.querySelector('#salary-step-inner'),
    assesmentInputs: document.querySelectorAll('[xname=inpvalorRateio]'),
    assesmentCostCenter: document.querySelectorAll('[xname=inpdescricaoCR]'),
    workloadShift: document.querySelectorAll('[xname=inpturnoIdentificacao]'),
    workloadStartHour: document.querySelectorAll('[xname=inpturnoHoraInicial]'),
    workloadEndHour: document.querySelectorAll('[xname=inpturnoHoraFinal]'),
    salaryProposed: document.querySelector('[xname=inpsalarioProposto]'),
    salaryBase: document.querySelector('[xname=inpsalarioBaseRef]'),
    maxPoint: document.querySelector('[xname=inppontoMaximoRef]')
  })

  return {
    mount
  }

  /**
   * 🔑 Métodos Públicos
   */

  /**
   * Realiza a montagem do módulo
   * @public
   * @memberof module:OverviewTabSalary
   */
  function mount () {
    $refs = getRefs()

    handleSalaryMessage()
    mountAssessmentBarChart()
    mountWorkloadTable()
    mountAssesmentTable()
    setStepSalaryPosition()
  }

  /**
   * 🔒 Métodos Privados
   */

  function handleSalaryMessage () {
    if (isSalaryAboveMidPoint()) {
      $refs.salaryAlertMessage
        .classList.remove('u-hidden')
    }
  }

  function mountAssessmentBarChart () {
    const segments = [
      ...$refs.assesmentInputs
    ].map((input, i) => {
      const barValue = parseFloat(input.value)
      const barLabel = $refs.assesmentCostCenter[i].value
      return getBarChartSegment(barValue, barLabel)
    })

    $refs.assesmentBarChart
      .insertAdjacentHTML(
        'afterbegin',
        segments.join('')
      )
  }

  function getBarChartSegment (value, label) {
    return `<div class="o-bar" style="width: ${value}%;" data-tooltip="${label}">${value}%</div>`
  }

  function mountAssesmentTable () {
    const assesmentValues = [...$refs.assesmentInputs]
      .map(input => `${input.value}%`)

    const columns = [
      { fields: $refs.assesmentCostCenter, cellTemplate: assesmentLabelTemplate },
      { values: assesmentValues, cellClass: 'u-text-right' }
    ]

    const tbody = $refs.assessmentTable.querySelector('tbody')
    const rows = createTableRowsByFields(columns)

    tbody.insertAdjacentHTML('afterbegin', rows)
  }

  function mountWorkloadTable () {
    const workloadPeriod = [...$refs.workloadStartHour]
      .map((input, i) =>
        `${input.value} às ${$refs.workloadEndHour[i].value}`
      )

    const columns = [
      { fields: $refs.workloadShift },
      { values: workloadPeriod, cellClass: 'u-text-right' }
    ]

    const tbody = $refs.workloadTable.querySelector('tbody')
    const rows = createTableRowsByFields(columns)

    tbody.insertAdjacentHTML('afterbegin', rows)
  }

  function setStepSalaryPosition () {
    const leftPos = getPercentDistance({
      min: $refs.salaryBase.value,
      max: $refs.maxPoint.value,
      current: currencyToNumber($refs.salaryProposed.value)
    })

    $refs.salaryStepInner.style.left = `${leftPos}%`
  }
}

export default OverviewTabSalary()
