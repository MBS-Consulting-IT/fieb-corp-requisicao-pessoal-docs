    import { showField, hideField } from '../../components/form-util/visibility'

/**
 * Controla as interações da seção `Vínculo Empregatício`
 * @module RequestContract
 */
const RequestContract = function () {
  const $refs = {
    contractCard: document.querySelector('#contract-details'),
    contractType: document.querySelectorAll('[xname=inptipoVinculo]')
  }

  const FormUtilConfig = {
    container: '.o-stack-input',
    hideClass: 'u-hidden'
  }

  /**
   * @memberof module:RequestContract
   * @property {object} contractsDetails - configuração de campos conforme tipo de contrato
   * @property {object} contractsDetails.long - configuração aplicada a vínculos de tempo determinado
   * @property {object} contractsDetails.short - configuração aplicada a vínculos de MOT
   * @property {string[]} contractsDetails.any.options - vínculos a serem considerados
   * @property {string[]} contractsDetails.any.fields - campos a serem habilitados
   */
  const state = {
    contractsDetails: {
      long: {
        options: ['Tempo Determinado', 'Horista CTD'],
        fields: [
          'duracao',
          'contratoProjeto',
          'dataFinal',
          'verbaProjeto'
        ]
      },
      short: {
        options: ['Mão de Obra Temporária'],
        fields: [
          'duracao',
          'justificativaMOT'
        ]
      }
    }
  }

  return {
    mount
  }

  /**
   * 🔑 Métodos Públicos
   */

  /**
   * Realiza a montagem do módulo
   * @public
   * @memberof module:RequestContract
   */
  function mount () {
    addTriggers()
    disabledAutocomplete()
    handleContractDetails()
  }

  /**
   * 🔒 Métodos Privados
   */

  function addTriggers () {
    $refs.contractType
      .forEach(field =>
        field.addEventListener('change', handleContractDetails)
      )
  }

  function disabledAutocomplete () {
    const { contractsDetails } = state

    const fields = [
      ...contractsDetails.long.fields,
      ...contractsDetails.short.fields
    ]

    fields.forEach(field => {
      document
        .querySelector(`[xname=inp${field}]`)
        .setAttribute('autocomplete', 'off')
    })
  }

  function handleContractDetails () {
    const type = getContract()

    const hasDetails = [
      ...state.contractsDetails.long.options,
      ...state.contractsDetails.short.options
    ].includes(type)

    hasDetails
      ? showContractCard(type)
      : hideContractCard()
  }

  function getContract () {
    const checked = [...$refs.contractType]
      .find(inp => inp.checked)

    return checked
      ? checked.value
      : null
  }

  function showContractCard (type) {
    const { visible, hidden } = getDetailsFields(type)
    const firstField = document.querySelector(`[xname=inp${visible[0]}]`)

    visible.forEach(field =>
      showField(field, FormUtilConfig)
    )

    hidden.forEach(field =>
      hideField(field, FormUtilConfig)
    )

    if (firstField) {
      firstField.focus()
    }

    $refs.contractCard.classList.add('-active')
  }

  function getDetailsFields (type) {
    const { contractsDetails } = state

    const isShort = contractsDetails.short.options.includes(type)
    const shortFields = contractsDetails.short.fields
    const isLong = contractsDetails.long.options.includes(type)
    const longFields = contractsDetails.long.fields

    let visible = []
    let hidden = []

    if (isShort) {
      visible = shortFields
      hidden = unique(longFields, visible)
    }

    if (isLong) {
      visible = longFields
      hidden = unique(shortFields, visible)
    }

    return { visible, hidden }
  }

  function hideContractCard () {
    const { contractsDetails } = state

    const fields = [
      ...contractsDetails.short.fields,
      ...contractsDetails.long.fields
    ]

    fields.forEach(field =>
      hideField(field, FormUtilConfig)
    )

    $refs.contractCard.classList.remove('-active')
  }
}

export default RequestContract()

const unique = (firstArray, secondArray) =>
  firstArray.filter(item =>
    secondArray.indexOf(item) === -1
  )
