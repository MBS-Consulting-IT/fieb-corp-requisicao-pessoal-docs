    import {
  isPositionForDisability,
  hasAccessibility
} from './overview-common'

/**
 * Realiza a montagem da aba `Cargo`
 *
 * @module OverviewTabPosition
 */
const OverviewTabPosition = function () {
  const $refs = {
    contractType: document.querySelector('[xname=inptipoVinculo]'),
    macroprocess: document.querySelector('[xname=inpMacroprocesso]'),
    contractDetails: document.querySelector('#contract-details-list'),
    internDetails: document.querySelector('#intern-details-list'),
    positionDetails: document.querySelector('#position-details-list'),
    macroprocessDetail: document.querySelector('dd[data-bind=Macroprocesso]'),
    workConditionsList: document.querySelector('#work-conditions-list'),
    acessibilityAlert: document.querySelector('#accessibility-alert'),
    workConditions: document.querySelectorAll('[xname="inpcodicoesTrabalho"]')
  }

  return {
    mount
  }

  /**
   * 🔑 Métodos Públicos
   */

  /**
   * Realiza a montagem do módulo
   * @public
   * @memberof module:OverviewTabPosition
   */
  function mount () {
    handleContractDetails()
    handleWorkConditions()
    handleInternDetails()
    handleMacroprocess()
  }

  /**
   * 🔒 Métodos Privados
   */

  function handleEmptyField (field) {
    const id = field.getAttribute('data-bind')
    const value = document.querySelector(`[xname=inp${id}]`).value
    const dd = field.closest('dd')
    const dt = dd.previousElementSibling

    if (value === '') {
      dd.remove()
      dt.remove()
    }
    return value
  }
  function handleContractDetails () {
    const values = [
      ...$refs.contractDetails.querySelectorAll('[data-bind]')
    ].map(handleEmptyField)

    const isAllEmpty = values.every(value => value === '')

    if (isAllEmpty) {
      $refs.contractDetails.remove()
    }
  }

  function handleWorkConditions () {
    const positionForDisability = isPositionForDisability()
    const unityHasNoAccessibility = !hasAccessibility()

    const conditions = [...$refs.workConditions]
      .filter(field => field.value)
      .map(field =>
        `<li><span class="o-chip">${field.value}</span></li>`
      )

    if (positionForDisability) {
      conditions.push(
        '<li><span class="o-chip"><span class="o-dot -pulse -orange">Vaga para PCD</span></span></li>'
      )
    }

    if (positionForDisability && unityHasNoAccessibility) {
      $refs.acessibilityAlert
        .classList.remove('u-hidden')
    }

    if (!conditions.length) {
      return $refs.workConditionsList
        .closest('.o-tab-container')
        .remove()
    }

    $refs.workConditionsList
      .insertAdjacentHTML(
        'afterbegin',
        conditions.join('')
      )
  }

  function handleInternDetails () {
    const isIntern = $refs.contractType.value === 'Estágio'

    return isIntern
      ? $refs.internDetails.classList.remove('u-hidden')
      : $refs.internDetails.classList.add('u-hidden')
  }

  function handleMacroprocess () {
    [
      ...$refs.positionDetails
        .querySelectorAll('[data-bind]')
    ].map(handleEmptyField)
  }
}

export default OverviewTabPosition()
